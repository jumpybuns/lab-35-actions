import { getPosts } from './postSelectors';

describe('Blog Selectors', () => {
  it('Selects all posts a post selector', () => {
    const state = {
      posts: [
        {
          title: 'title Post',
          body: 'body',
        },
        {
          title: 'title Post2',
          body: 'body2',
        },
      ],
    };

    const posts = getPosts(state);

    expect(posts).toEqual([
      {
        title: 'title Post',
        body: 'body',
      },
      {
        title: 'title Post2',
        body: 'body2',
      },
    ]);
  });
});
